import schedule
import time
from datetime import datetime

def monitor_all_regions():
    """Run monitoring for all Lake Victoria regions"""
    regions = ['tanzanian', 'kenyan', 'ugandan', 'central']
    
    for region in regions:
        print(f"[{datetime.now()}] Monitoring Lake Victoria - {region}")
        subprocess.run([
            "python", "src/main.py",
            "--region", region,
            "--quick"
        ])
        print(f"âœ“ Completed {region}")

def monitor_other_lakes():
    """Run monitoring for other lakes"""
    lakes = ['tanganyika', 'malawi', 'kyoga']
    
    for lake in lakes:
        print(f"[{datetime.now()}] Monitoring {lake.title()}")
        subprocess.run([
            "python", "src/main.py",
            "--lake", lake,
            "--quick"
        ])

# Schedule daily monitoring
schedule.every().day.at("02:00").do(monitor_all_regions)
schedule.every().day.at("03:00").do(monitor_other_lakes)

# Keep scheduler running
while True:
    schedule.run_pending()
    time.sleep(60)
